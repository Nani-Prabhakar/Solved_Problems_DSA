class Solution {
  public:
    bool canPair(vector<int> nums, int k) {
        // Code here.
          if(nums.size()%2!=0) {
            return false;
        }
        vector<int>mp(k+1,0);
        for(int i=0;i<nums.size();i++) {
            int x = nums[i] % k;
            mp[x]++;
            mp[k-x]--;
        }
        for(int i=1;i<k;i++) {
            if(mp[i]!=0) {
                return false;
            }
        }
        return true;
    }
};
