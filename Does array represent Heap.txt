class Solution{
    public:
    bool isMaxHeap(int arr[], int n)
    {
        // Your code goes here
        for(int i=0;i<n;i++){
            int child1=2*i+1;
            int child2=2*i+2;
            if(child1<n){
                if(arr[i]<arr[child1])return 0;
            }
            if(child2<n){
                if(arr[i]<arr[child2])return 0;
            }
        }
        return 1;
    }
};
